\begin{algorithm}[!htbp]
  \caption{硬币问题}
  \begin{algorithmic}[1]
  \Procedure{DP}{$n, v, w, y$}
    \For {$i \gets 0 \text{ to } y$}
      \State $dp[i] \gets \infty$
      \State $s[i] \gets 0$
    \EndFor

    \State $dp[0] \gets 0$
    \For {$i \gets 1 \text{ to } n$}
      \For {$j \gets v_i \text{ to } y$}
        \If {$dp[j - v_i] + w_i < dp[j] $}
          \State $dp[j] \gets dp[j - v_i] + w_i$
          \State $s[j] \gets i$
        \EndIf
      \EndFor
    \EndFor
  \EndProcedure
  \end{algorithmic}
\end{algorithm}

database <- read.csv('data.csv')
head(database)
cor(database)

lm.sol <- lm(age ~ x_l + x_d + x_h + x_whole + x_shucked + x_viscera
  + x_shell, data=database)

summary(lm.sol)

pdf("lm.sol.pdf")
plot(lm.sol)

lm.step = step(lm.sol)
summary(lm.step)

pdf("lm.step.pdf")
plot(lm.step)
